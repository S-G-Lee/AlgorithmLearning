def exhaustive_search(N, B, heights, i, tot):
    global result

    if i == N:
        if tot >= B:
            if result > tot - B:
                result = tot - B
        return
    elif tot >= B:
        if result > tot - B:
                result = tot - B
        return

    exhaustive_search(N, B, heights, i+1, tot + heights[i])
    exhaustive_search(N, B, heights, i+1, tot)
